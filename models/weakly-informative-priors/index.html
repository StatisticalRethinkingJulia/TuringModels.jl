Data

Model

using Turing

@model function m8_3(y)
    α ~ Normal(1, 10)
    σ ~ truncated(Cauchy(0, 1), 0, Inf)
    μ = α

    y ~ Normal(μ, σ)
end;

Output

chns = sample(m8_3(y), NUTS(), 1000)

using StatsPlots

StatsPlots.plot(chns)
